		FROM printer pt
        INNER JOIN product AS pr ON pt.model = pr.model
			WHERE pr.maker LIKE('%Lenovo%')
UNION SELECT pr.model, pc.price
		FROM pc
        INNER JOIN product AS pr ON pc.model = pr.model
			WHERE pr.maker LIKE('%Lenovo%')
UNION SELECT pr.model, lp.price
		FROM laptop lp
        INNER JOIN product AS pr ON lp.model = pr.model
			WHERE pr.maker LIKE('%Lenovo%')

	FROM product pr
	left JOIN (SELECT pr.maker as maker, COUNT(*) as count
			FROM product pr
			WHERE pr.type LIKE('Laptop') 
			group by pr.maker 	) as A2 ON pr.maker = A2.maker
		WHERE pr.type LIKE('PC') and A2.count IS NULL

SELECT pr.maker
		FROM pc
			INNER JOIN  product pr ON pc.model = pr.model
            LEFT JOIN ( SELECT DISTINCT pr.maker as maker
					FROM laptop lap
						INNER JOIN product as pr ON lap.model = pr.model
                        WHERE lap.speed >= 750
						) A1 ON pr.maker = A1.maker
				WHERE pc.speed >= 750 and pr.maker = A1.maker

SELECT A1.model, A1.max
		FROM (	SELECT pr.model as model, MAX(pc.price) as max
					FROM product pr, pc
						WHERE pr.model = pc.model
				UNION SELECT pr.model as model, MAX(lp.price) as max
					FROM product pr
					INNER JOIN laptop AS lp ON pr.model = lp.model
				UNION SELECT pr.model as model, MAX(pt.price) as max
					FROM product pr	
					INNER JOIN printer as pt ON pt.model = pr.model
				) A1 
	INNER JOIN (
		SELECT MAX(A1.max) as maxx
			FROM(
				SELECT pr.model as model, MAX(pc.price) as max
					FROM product pr, pc
						WHERE pr.model = pc.model
				UNION SELECT pr.model as model, MAX(lp.price) as max
					FROM product pr
					INNER JOIN laptop AS lp ON pr.model = lp.model
				UNION SELECT pr.model as model, MAX(pt.price) as max
					FROM product pr	
					INNER JOIN printer as pt ON pt.model = pr.model
				) A1 
		) A2 
        WHERE A1.max = A2.maxx

	FROM( SELECT pc.model as model, pc.ram as ram, pc.speed as speed
			FROM pc
			) A1
	INNER JOIN ( SELECT MIN(pc.ram) as min
					FROM pc
                    ) A2 
	INNER JOIN ( SELECT MAX(pc.speed) as max
				FROM pc
                	INNER JOIN ( SELECT MIN(pc.ram) as min
					FROM pc
                    ) A2
				WHERE pc.ram = A2.min)
                    A3
			WHERE A1.ram = A2.min and A1.speed = A3.max
